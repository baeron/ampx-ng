import { Injectable } from '@angular/core';
import { Http, Headers } from '@angular/http';
import 'rxjs/add/operator/map';
// tslint:disable-next-line:import-blacklist
import { Observable } from 'rxjs';
import 'rxjs/add/operator/map';
import 'rxjs/add/operator/mergeMap';

@Injectable()
export class CableService {
  headers: Headers = new Headers();

  constructor(private http: Http) { }

  getObjectsById(id, arrayList, queryString) {
    const observableBatch = [];

    arrayList.forEach((key) => {
      // for dev
      observableBatch.push(this.http.get('http://localhost:3000/project/' + id + '/' + queryString + '/' + key).map((res) => res.json()));
      // for deploy
      /*
      observableBatch.push(this.http.get('project/' + id + '/' + queryString + '/' + key).map((res) => res.json()));
      */
    });
    return Observable.forkJoin(observableBatch);
  }

  getCabelExcelList(id, array, queryString): Observable<any> {
    return Observable.forkJoin(
      this.getObjectsById(id, array, queryString)
    );
  }

  getCables(id) {
    const headers = new Headers();
    this.headers.append('Content-Type', 'application/json');
    // for dev use
    return this.http.get('http://localhost:3000/project/' + id + '/cables', {headers: headers})
      .map(res => res.json());
    // for deploy
  /*
    return this.http.get('project/' + id + '/cables', { headers: this.headers })
      .map(res => res.json());
      */
  }

  getElectricalName(id) {
    const headers = new Headers();
    headers.append('Content-Type', 'application/json');
    // for dev use
    return this.http.get('http://localhost:3000/project/' + id + '/electrical', {headers: headers})
      .map(res => res.json());
    // for deploy
  /*
    return this.http.get('project/' + id + '/electrical', { headers: headers })
      .map(res => res.json());
      */
  }

  createCable(id, cable) {
    const headers = new Headers();
    headers.append('Content-Type', 'application/json');
    // for dev use
    /*
        return this.http.post('http://localhost:3000/project/'+id+'/cable-create/', cable, {headers: headers})
          .map(res => res.json());
    */
    // for deploy

    return this.http.post('project/' + id + '/cable-create/', cable, { headers: headers })
      .map(res => res.json());

  }

  getCableItem(projectId, cableId) {
    const headers = new Headers();
    headers.append('Content-Type', 'application/json');
    // for dev use
    return this.http.get('http://localhost:3000/project/' + projectId + '/cables/' + cableId, {headers: headers})
      .map(res => res.json());
    // for deploy
    /*
    return this.http.get('project/' + projectId + '/cables/' + cableId, { headers: headers })
      .map(res => res.json());
      */
  }

  updateCabelItem(projectId, cabelId, cabelItem) {
    const headers = new Headers();
    headers.append('Content-Type', 'application/json');
    // for dev use
    return this.http.patch('http://localhost:3000/project/' + projectId + '/cable-update/' + cabelId, cabelItem, {headers: headers})
      .map(res => res.json());
    // for deploy
    /*
    return this.http.patch('project/' + projectId + '/cable-update/' + cabelId, cabelItem, { headers: headers })
      .map(res => res.json());
      */
  }

  deleteCableItem(projectId, cableId) {
    const headers = new Headers();
    headers.append('Content-Type', 'application/json');
    // for dev use
    return this.http.delete('http://localhost:3000/project/' + projectId + '/cables/' + cableId, {headers: headers})
      .map(res => res.json());
    // for deploy
    /*
    return this.http.delete('project/' + projectId + '/cables/' + cableId, { headers: headers })
      .map(res => res.json());
      */
  }
}
