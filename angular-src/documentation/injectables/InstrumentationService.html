import { Injectable } from '@angular/core';
import { Http, Headers } from '@angular/http';
import 'rxjs/add/operator/map';
// tslint:disable-next-line:import-blacklist
import { Observable } from 'rxjs';
import 'rxjs/add/operator/map';
import 'rxjs/add/operator/mergeMap';

@Injectable()
export class InstrumentationService {

  constructor(private http: Http) { }

  getInstrumentationsList(id) {
    const headers = new Headers();
    headers.append('Content-Type', 'application/json');
    // for dev use
    
          return this.http.get('http://localhost:3000/project/'+id+'/instrumentations-list', {headers: headers})
            .map(res => res.json());
    
    // for deploy
    /*
    return this.http.get('project/' + id + '/instrumentations-list', { headers: headers })
      .map(res => res.json());
      */
  }

  getInstrumentations(id) {
    const headers = new Headers();
    headers.append('Content-Type', 'application/json');
    // for dev use
    
        return this.http.get('http://localhost:3000/project/' + id + '/instrumentations', {headers: headers})
          .map(res => res.json());
    
    // for deploy
    /*
    return this.http.get('project/' + id + '/instrumentations', { headers: headers })
      .map(res => res.json());
      */
  }

  createInstrumentstion(id, instrumentation) {
    const headers = new Headers();
    headers.append('Content-Type', 'application/json');
    // for dev use
    
        return this.http.post('http://localhost:3000/project/'+id+'/instrumentation-create/', instrumentation, {headers: headers})
          .map(res => res.json());
    
    // for deploy
  /*
    return this.http.post('project/' + id + '/instrumentation-create/', instrumentation, { headers: headers })
      .map(res => res.json());
      */
  }

  getInstrumentationItem(projectId, instrumentationId) {
    const headers = new Headers();
    headers.append('Content-Type', 'application/json');
    // for dev use
    
        return this.http.get('http://localhost:3000/project/'+projectId+'/instrumentations/'+instrumentationId, {headers: headers})
          .map(res => res.json());
    
    // for deploy
    /*
    return this.http.get('project/' + projectId + '/instrumentations/' + instrumentationId, { headers: headers })
      .map(res => res.json());
      */
  }

  getObjectsById(id, arrayList, queryString) {
    const observableBatch = [];

    arrayList.forEach((key) => {
      // for dev
      // observableBatch.push(this.http.get('http://localhost:3000/project/'+id+'/'+queryString+'/'+key).map((res) => res.json()));
      // for deploy
      observableBatch.push(this.http.get('project/' + id + '/' + queryString + '/' + key).map((res) => res.json()));
    });
    return Observable.forkJoin(observableBatch);
  }

  getInstrumentationsCheckList2(id, array, queryString): Observable<any> {
    return Observable.forkJoin(
      this.getObjectsById(id, array, queryString)
    );
  }

  updateInstrumentationItem(projectId, instrumentationId, instrumentationItem) {
    const headers = new Headers();
    headers.append('Content-Type', 'application/json');
    // for dev use
    
    return this.http.patch('http://localhost:3000/project/'+ projectId+'/instrumentation-update/'
    +instrumentationId, instrumentationItem, {headers: headers})
          .map(res => res.json());
    
    // for deploy
    /*
    return this.http.patch('project/' + projectId + '/instrumentation-update/' + instrumentationId, instrumentationItem, { headers: headers })
      .map(res => res.json());
      */
  }

  deleteInstrumentationItem(projectId, instrumentationId) {
    const headers = new Headers();
    headers.append('Content-Type', 'application/json');
    // for dev use
    
        return this.http.delete('http://localhost:3000/project/' + projectId + '/instrumentations/' + instrumentationId, {headers: headers})
          .map(res => res.json());
    
    // for deploy
    /*
    return this.http.delete('project/' + projectId + '/instrumentations/' + instrumentationId, { headers: headers })
      .map(res => res.json());
      */
  }
}
